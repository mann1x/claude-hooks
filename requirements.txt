# claude-hooks runtime deps (core).
#
# The core framework (hooks, MCP client, recall/store) is stdlib-only.
# The proxy module requires httpx + h2 so the forwarder can present an
# HTTP/2 connection profile to api.anthropic.com — stdlib http.client
# (HTTP/1.1-per-request) trips Anthropic's edge 429 gate.
#
# install.py pip-installs this file into the conda env when creating
# it, and re-installs when proxy.enabled is true and the deps are
# missing (see _ensure_proxy_deps in install.py).
#
# OPTIONAL provider extras live in their own files — install only the
# ones you actually enable in config/claude-hooks.json:
#
#   pip install -r requirements.txt                        # core only
#   pip install -r requirements.txt -r requirements-pgvector.txt
#   pip install -r requirements.txt -r requirements-sqlite-vec.txt
#
# Or all of them at once:
#   pip install -r requirements.txt -r requirements-pgvector.txt -r requirements-sqlite-vec.txt
#   pip install -r requirements-dev.txt                    # tests
#

httpx[http2]>=0.27
# ruff is invoked by the PostToolUse hook to lint Python files right
# after Edit/Write/MultiEdit, injecting diagnostics into the next
# turn's additionalContext. Ships in the core install because the
# hook is enabled by default. The hook gracefully degrades if ruff
# is missing on the host (logs a warning and skips), but new envs
# get it pre-installed via this requirements line.
ruff>=0.6
# tomli on Python <3.11 for claude_hooks.lsp_engine — the per-project
# .claude-hooks/lsp-engine.toml needs a TOML parser. 3.11+ uses the
# stdlib tomllib, no extra dep needed.
tomli>=2.0; python_version < "3.11"
